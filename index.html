						PSEUDOCODIGO MERGESORT 

							Funcion merge(A[],inicio, fin)											
							|	Si(inicio < fin)															
							|	|	med <- (inicio + fin)/2											     ESTUDIANTES: 
							|	|	merge(inicio, med)												-> JOSE LUIS VARILLAS MELENDEZ											
							|	|	merge(med+1, fin)												-> YAMILE CONCEPCION SILVA JARAMILLO								
							|	|	insertar <- inicio												-> ANDRES GABRIEL SALCEDO PATRONI
							|	|	izq <- inicio													-> ALESSANDRO GREGORY VALENZUELA MARTINEZ
							|	|	der <- med+1 													-> ERNESTO ZENOBIO BARRIAS PEÃ‘A
							|	|	Mientras (insertar <= fin)	
							|	|	|	Si (izq  fin o A[izq ] <= A[der])
							|	|	|	|	AUX[insertar] <- A[izq ]
							|	|	|	|	insert <- insert + 1 
							|	|	|	|	izq <- izq +1
							|	|	|	Sino 
							|	|	|	|	AUX[insertar] <- A[der]
							|	|	|	|	insert <- insert + 1
							|	|	|	|	der <- der +1 
							|	|	|	Fin_Si
							|	|	Fin_Mientras
							|	|	Para insertar desde inicio hasta fin
							|	|	|	A[insertar] <- AUX[insert]						:
							|	|	Fin_Para											   
							|	Sino 													  
							|	|	break												
							|	Fin_Si														
							Fin_Funcion
